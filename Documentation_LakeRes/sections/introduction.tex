\begin{pythoncode}[Une petite classe Python]
class Compteur:
    def __init__(self, start=0):
        self.count = start        # <- 'self' est coloré
    def increment(self):
        self.count += 1
        print(f"Valeur : {self.count}")

# Usage
if __name__ == "__main__":
    c = Compteur()
    for _ in range(3):
        c.increment()
\end{pythoncode}
