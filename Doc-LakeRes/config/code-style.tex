%==============================================================================
% CONFIGURATION POUR L'AFFICHAGE DU CODE AVEC MINTED
%==============================================================================
% Ce fichier contient la configuration pour l'affichage du code source
% Il configure les styles visuels et les environnements de code
%==============================================================================
% Configuration de base pour tous les blocs de code
\setminted{
  fontsize=\small,
  breaklines=true,
  autogobble=true,
  linenos=true,
  numbersep=5pt,
  frame=lines,
  framesep=2mm,
  style=monokai
}

% Définir un style Python spécifique
\setminted[python]{
  python3=true,
  tabsize=4,
  bgcolor=gdbg,
  fontfamily=tt,
  fontsize=\footnotesize,
  numbersep=12pt
}

%------------------------------------------------------------------------------
% CODE EN LIGNE AVEC MINTED - TEXTE COLORÉ SUR FOND GRIS
%------------------------------------------------------------------------------
% Définir les couleurs personnalisées
\definecolor{codeOrange}{HTML}{E8912D}  % Orange pour le texte
\definecolor{codeBg}{HTML}{F3F3F3}      % Gris clair pour le fond

% Version avec texte orange sur fond gris
\newcommand{\inlinecode}[1]{%
  \mintinline[bgcolor=codeBg, fontsize=\footnotesize, formatcom=\color{codeOrange}]{text}{#1}%
}
%------------------------------------------------------------------------------
% CHEMINS DE FICHIERS STYLISÉS
%------------------------------------------------------------------------------
% Chemin de fichier simple
\newcommand{\filepath}[1]{%
  \colorbox{pathBg}{%
    \mintinline[fontsize=\small,bgcolor=]{text}{#1}%
  }%
}

% Chemin avec icône de dossier
\newcommand{\folderpath}[1]{%
  \colorbox{pathBg}{%
    \textcolor{folderColor}{\faFolder}\,%
    \mintinline[fontsize=\small,bgcolor=]{text}{#1}%
  }%
}

% Chemin URL
\newcommand{\urlpath}[1]{%
  \colorbox{urlBg}{%
    \textcolor{linkColor}{\faLink}\,%
    \mintinline[fontsize=\small,bgcolor=]{text}{#1}%
  }%
}

%------------------------------------------------------------------------------
% STYLES ADDITIONNELS POUR DIFFÉRENTS LANGAGES
%------------------------------------------------------------------------------
% Style pour bash/shell
\setminted[bash]{
  style=monokai,
  fontsize=\footnotesize
}

% Style pour HTML
\setminted[html]{
  style=monokai,
  fontsize=\footnotesize
}

% Style pour JavaScript
\setminted[javascript]{
  style=monokai,
  fontsize=\footnotesize
}

%------------------------------------------------------------------------------
% ENVIRONNEMENTS RACCOURCIS POUR DIFFÉRENTS LANGAGES
%------------------------------------------------------------------------------
% Environnement facile pour le code bash
\newtcblisting{bashcode}[1][]{
  pythonbox,
  minted language=bash,
  title=#1
}

% Environnement facile pour le code HTML
\newtcblisting{htmlcode}[1][]{
  pythonbox,
  minted language=html,
  title=#1
}

% Environnement facile pour le code JavaScript
\newtcblisting{jscode}[1][]{
  pythonbox,
  minted language=javascript,
  title=#1
}
